The Code

The Javascript Starts with the function getNumbers which retrieves the values from the web form using documnet.getElementById

To ensure and validate the values are integers, the inputs are parsed with parseInt and following with a conditional to check value type using Number.isInteger

If the input returns TRUE the numbers are generated and then displayed using the functions generateNumbers and displayNumbers respectively otherwise an alert is returned





The generateNumbers functions takes in both values and generates values for an empty list. The numbers are iterated over and pushed to the empty list which is then returned







To display the numbers to our app with displayNumbers the function takes the numbers array returned from the previous funtion. These numbers are then put through another loop that creates a template literal for generating the table rows.







After each iteration, the method document.getElementById("results").innerHTML and assigning the values for each literal


              function getNumbers() {
                let startNumber = document.getElementById("startNumber").value;
                let endNumber = document.getElementById("endNumber").value;
              
                startNumber = parseInt(startNumber);
                endNumber = parseInt(endNumber);
              
                if (Number.isInteger(startNumber) && Number.isInteger(endNumber)) {
                  let numbers = generateNumbers(startNumber, endNumber);
              
                  displayNumbers(numbers);
              
                } else {
                  alert("You Must Enter A Valid Integer");
                }
              }
              
              function generateNumbers(startNumber, endNumber) {
                let numbers = [];
              
                for (let number = startNumber; number <= endNumber; number++) {
                  numbers.push(number);
                }
              
                return numbers;
              }
              
              function displayNumbers(numbers) {
                let templateRows = "";
                for (let number = 0; number < numbers.length; number++) {
                  let displayNumber = numbers[number];
              
                  if (displayNumber % 2 === 0) {
                    templateRows += `${displayNumber}`;
                  } else {
                    templateRows += `${displayNumber}`;
                  }
              
                }
              
                document.getElementById("results").innerHTML = templateRows;
              
              }
